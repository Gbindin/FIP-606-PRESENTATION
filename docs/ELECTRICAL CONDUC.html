  mutate_at(vars(1:3), as.factor) |>
  mutate(SOWDATE = case_when(SOWDATE == "SD 1" ~ "Aug-09",

                             .default = SOWDATE)) -> dat

#Firstly, subject to ANOVA to determine if variety and sowing date had interaction effect.

library(emmeans)
library(multcomp)
library(multcompView)

model_CONDUC <- aov(CONDUC ~ VARIETY*SOWDATE + REP, data = dat)
summary(model_CONDUC)
